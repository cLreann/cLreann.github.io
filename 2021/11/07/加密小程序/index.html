加密小程序

c
加解密
发布日期:   2021-11-07 
#include <stdio.h>
#include <string.h> //提供一些字符串用到的函数
#include <stdlib.h> //系统函数
#include <ctype.h>

char ch = '0';
char filename[256]="";   //定义一个数组储存文件名
FILE *fp = NULL;		//定义一个指针  后面用于指向打开的文件
FILE *fptemp = NULL;	//定义一个指针   后面用于指向生成的临时文件（加密或解密后的内容）
char password[12] = "123456";
const char tempfile[256] = "temp123456.temp"; //const 定义一个常量
int i =0;		//定义一个循环控制变量
int pwdlen = 0;	//定义一个记录密码长度的整形变量


//函数声明
void menu();
void inputpass(char *pass);
//函数实现
void inputpass(char *pass)
{
    scanf("%s",pass);
}

void menu()
{
    printf("************************************\n");
    printf("************************************\n");

    //1:打开文件或目录
    //通过字符终端读取一个字符串
    printf("清楚如要加密或要解密的文件路径,例如/user/go/1.txt \n");
    gets(filename);
    if(NULL ==(fp = fopen(filename,"rb")))
    {
        printf("你所输入的文件不存在\n");
        exit(1);
    }
    //如果文件存在
    printf("文件存在，请输入密码。如：8888888\n");
    inputpass(password);
    pwdlen = (int)strlen(password); //获取密码的长度
    if(0 == pwdlen)
    {
        printf("对不起，密码长度不能为0\n");
        exit(2);
    }
    //2：读出文件中的内容进行加密//3：把加密的信息写入文章中并覆盖
    fptemp = fopen(tempfile,"wb");
    while(1)
    {
        ch = getc(fp);   //getc用于从流中取字符，若从一个文件中读取一个字符，读到文件尾而无数据时便返回EOF
        if(feof(fp))	//feof标准库函数,在stdio.h中，检测流上的文件结束符，如文件结束，返回非0值，否则返0
        {
            /*判断文件读完没有*/
            break;
        }
        /*每取出一个字符就加密*/
        ch^= password[i++];     // a^b 表示a异或b  a^= b a与b异或结果给a
        //ch里就是加密以后的数据了
        fputc(ch,fptemp);		//写入
        if(i == pwdlen)
        {
            i=0;
        }
    }

    //循环结束后清理
    fclose(fp);
    fclose(fptemp);
    remove(filename);
    rename(tempfile,filename);
    printf("恭喜你，加密或解密成功\n");
}

int main(int argc, char const *argv[])
{
    menu();
    return 0;
}
